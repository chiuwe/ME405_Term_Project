00058 /** \brief This method takes one A/D reading from the given channel and returns it. 
00059  *  \details \b Details: This method selects a channel to read and goes through the A/D conversion.
00060  *  This method also guards against time-out.
00061  *  @param  ch The A/D channel which is being read must be from 0 to 7.
00062  *  @return The result of the A/D conversion.
00063  */
00064 
00065 uint16_t adc::read_once (uint8_t ch) {
00066    uint8_t i = 0;
00067    uint16_t result;
00068 
00069    ADMUX = ADMUX_init;
00070    ADMUX |= ch;
00071 
00072    // start conversion with ADSC
00073    ADCSRA |= (1 << ADSC);
00074    while (ADCSRA & (1 << ADSC) && i < 65) {
00075       i++;
00076    }
00077    
00078    result = ADCL | (ADCH << 8);
00079    //DBG (ptr_to_serial, "result: " << result << endl);
00080    return i == 65 ? -1 : result;
00081 }

00094 uint16_t adc::read_oversampled (uint8_t channel, uint8_t samples) {
00095     uint16_t sum = 0;
00096    uint8_t i;
00097 
00098    if (samples > 32) {
00099       samples = 32;
00100    }
00101    
00102    for (i = 0; i < samples; i++) {
00103       sum += read_once(channel);

00106    //DBG (ptr_to_serial, "average: " << sum / samples << endl);

00062 void task_motor::run (void) {
00063    uint16_t a2d_reading;
00064 
00065    adc *p_my_adc = new adc(p_serial);
00066    PORTC |= (1 << 3) | (1 << 4);
00067    
00068    // This is the task loop for the motor control task. This loop runs until the
00069    // power is turned off or something equally dramatic occurs.
00070    for (;;) {
00071       if (PINC & (1 << brake_pin) || brake->get()) {
00072          driver->brake();
00073       } else {
00074          if (pot->get()) {
00075             a2d_reading = p_my_adc->read_once(adc_select);
00076             driver->set_power((a2d_reading / 2) - 255);
00077          } else {
00078             driver->set_power(power->get());
00079          }
00080       }
00081    }
00082    runs++;
00083    // TODO: delay might need to be longer. task priority not working if different.
00084    delay (100);
